#!/bin/bash

if [ $# -ne 1 ]; then
	echo "Error: se requiere 1 parametro"
	exit 1
fi

nombre=$1

if [ -e "$nombre" ]; then
	if [ -f "$nombre" ]; then
		echo "'$nombre' es un archivo."
	elif [ -d "$nombre" ]; then
		echo "'$nombre' es un directorio."
	fi
else
	echo "'$nombre' no existe. Se creara un directorio con ese nombre."
	mkdir "$nombre"
	echo "Directorio '$nombre' creado."
fi
